# 2043. Simple Bank System

## Solution 1

```python
class Bank(object):

    def __init__(self, balance):
        """
        :type balance: List[int]
        """
        self.balance = balance
        self.cnt = len(balance)

    def transfer(self, account1, account2, money):
        """
        :type account1: int
        :type account2: int
        :type money: int
        :rtype: bool
        """
        if account1 < 1 or account1 > self.cnt:
            return False
        if account2 < 1 or account2 > self.cnt:
            return False
        if self.balance[account1 - 1] < money:
            return False
        self.balance[account1 - 1] -= money
        self.balance[account2 - 1] += money
        return True

    def deposit(self, account, money):
        """
        :type account: int
        :type money: int
        :rtype: bool
        """
        if account < 1 or account > self.cnt:
            return False
        self.balance[account - 1] += money
        return True

    def withdraw(self, account, money):
        """
        :type account: int
        :type money: int
        :rtype: bool
        """
        if account < 1 or account > self.cnt:
            return False
        if self.balance[account - 1] < money:
            return False
        self.balance[account - 1] -= money
        return True
```
