# 3823. Reverse Letters Then Special Characters in a String

## Solution 1 (time O(n), space O(n))

```python
class Solution:
    def reverseByType(self, s: str) -> str:
        n = len(s)
        s = list(s)
        p0, p1 = 0, n - 1
        while p0 < p1:
            while p0 < p1 and not s[p0].isalpha():
                p0 += 1
            while p0 < p1 and not s[p1].isalpha():
                p1 -= 1
            if p0 < p1:
                s[p0], s[p1] = s[p1], s[p0]
                p0 += 1
                p1 -= 1
        p0, p1 = 0, n - 1
        while p0 < p1:
            while p0 < p1 and s[p0].isalpha():
                p0 += 1
            while p0 < p1 and s[p1].isalpha():
                p1 -= 1
            if p0 < p1:
                s[p0], s[p1] = s[p1], s[p0]
                p0 += 1
                p1 -= 1
        return "".join(s)
```
