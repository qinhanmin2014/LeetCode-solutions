# 3830. Longest Alternating Subarray After Removing At Most One Element

## Solution 1 (time O(n), space O(n))

```python
# Dynamic programming
class Solution:
    def longestAlternating(self, nums: List[int]) -> int:
        n = len(nums)
        left = [[0] * 2 for _ in range(n)]
        right = [[0] * 2 for _ in range(n)]
        left[0][0] = left[0][1] = 1
        for i in range(1, n):
            if nums[i] > nums[i - 1]:
                left[i][0] = 1
                left[i][1] = left[i - 1][0] + 1
            elif nums[i] < nums[i - 1]:
                left[i][0] = left[i - 1][1] + 1
                left[i][1] = 1
            else:
                left[i][0] = left[i][1] = 1
        right[n - 1][0] = right[n - 1][1] = 1
        for i in range(n - 2, -1, -1):
            if nums[i + 1] > nums[i]:
                right[i][0] = 1
                right[i][1] = right[i + 1][0] + 1
            elif nums[i + 1] < nums[i]:
                right[i][0] = right[i + 1][1] + 1
                right[i][1] = 1
            else:
                right[i][0] = right[i][1] = 1
        ans = 0
        for i in range(n):
            for j in range(2):
                ans = max(ans, left[i][j])
        for i in range(1, n - 1):
            if nums[i + 1] > nums[i - 1]:
                ans = max(ans, left[i - 1][0] + right[i + 1][0])
            elif nums[i + 1] < nums[i - 1]:
                ans = max(ans, left[i - 1][1] + right[i + 1][1])
        return ans
```
