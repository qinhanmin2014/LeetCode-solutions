# 3829. Design Ride Sharing System

## Solution 1 (time O(1), space O(n))

```python
class RideSharingSystem:

    def __init__(self):
        self.riders = deque()
        self.drivers = deque()
        self.waiting_riders = set()

    def addRider(self, riderId: int) -> None:
        self.riders.append(riderId)
        self.waiting_riders.add(riderId)

    def addDriver(self, driverId: int) -> None:
        self.drivers.append(driverId)

    def matchDriverWithRider(self) -> List[int]:
        while self.riders and self.riders[0] not in self.waiting_riders:
            self.riders.popleft()
        if not self.riders or not self.drivers:
            return [-1, -1]
        cur_rider = self.riders.popleft()
        cur_driver = self.drivers.popleft()
        self.waiting_riders.discard(cur_rider)
        return [cur_driver, cur_rider]

    def cancelRider(self, riderId: int) -> None:
        self.waiting_riders.discard(riderId)
```
