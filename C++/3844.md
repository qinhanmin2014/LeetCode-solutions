# 3844. Longest Almost-Palindromic Substring

## Solution 1 (time O(n^2), space O(n^2))

```cpp
// Dynamic programming
class Solution {
public:
    int almostPalindromic(string s) {
        int n = s.length();
        vector<vector<bool>> dp0(n, vector<bool>(n, false));
        vector<vector<bool>> dp1(n, vector<bool>(n, false));
        for (int i = 0; i < n; i++) {
            dp0[i][i] = true;
            dp1[i][i] = true;
        }
        for (int i = 0; i < n - 1; i++) {
            if (s[i] == s[i + 1]) {
                dp0[i][i + 1] = true;
                dp1[i][i + 1] = true;
            } else {
                dp1[i][i + 1] = true;
            }
        }
        for (int l = 3; l <= n; l++) {
            for (int i = 0; i <= n - l; i++) {
                int j = i + l - 1;
                if (s[i] == s[j]) {
                    if (dp0[i + 1][j - 1]) {
                        dp0[i][j] = true;
                    }
                    if (dp1[i + 1][j - 1]) {
                        dp1[i][j] = true;
                    }
                } else {
                    if (dp0[i + 1][j] || dp0[i][j - 1]) {
                        dp1[i][j] = true;
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (dp1[i][j]) {
                    ans = max(ans, j - i + 1);
                }
            }
        }
        return ans;
    }
};
```
